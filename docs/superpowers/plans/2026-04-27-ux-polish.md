**Goal:** Add global Toast notifications, suggested prompts for new users, and lazy-load heavy components to reduce first-paint JS by ~350KB.

**Architecture:** Three independent subsystems — a Zustand-backed toast notification system mounted globally in ClientProviders, a suggested prompts component rendered in the idle chat state, and Next.js dynamic imports for MapPanel/DynamicIsland. Each can be implemented and tested independently.

**Tech Stack:** Zustand (store), framer-motion (animations), next/dynamic (code splitting), lucide-react (icons)

```tsx
"use client"
import { create } from "zustand"
import { motion, AnimatePresence } from "framer-motion"
import { Check, X, Info, AlertTriangle } from "lucide-react"
import { useEffect, useCallback } from "react"

type ToastType = "success" | "error" | "info" | "warning"

interface Toast {
  id: string
  message: string
  type: ToastType
  createdAt: number
}

interface ToastStore {
  toasts: Toast[]
  addToast: (message: string, type?: ToastType, duration?: number) => void
  removeToast: (id: string) => void
}

let toastCounter = 0

export const useToastStore = create<ToastStore>((set, get) => ({
  toasts: [],
  addToast: (message, type = "info", duration = 3000) => {
    const id = `toast-${++toastCounter}`
    const now = Date.now()
    // Deduplicate: skip if same message was added within 2s
    const recent = get().toasts.find(
      (t) => t.message === message && now - t.createdAt < 2000
    )
    if (recent) return
    set((s) => ({ toasts: [...s.toasts, { id, message, type, createdAt: now }] }))
    if (duration > 0) {
      setTimeout(() => get().removeToast(id), duration)
    }
  },
  removeToast: (id) => set((s) => ({ toasts: s.toasts.filter((t) => t.id !== id) })),
}))

const ICONS: Record<ToastType, React.ReactNode> = {
  success: <Check className="h-3.5 w-3.5" />,
  error: <X className="h-3.5 w-3.5" />,
  info: <Info className="h-3.5 w-3.5" />,
  warning: <AlertTriangle className="h-3.5 w-3.5" />,
}

const COLORS: Record<ToastType, { border: string; text: string; bg: string }> = {
  success: { border: "border-emerald-500/30", text: "text-emerald-400", bg: "bg-emerald-500/5" },
  error: { border: "border-red-500/30", text: "text-red-400", bg: "bg-red-500/5" },
  info: { border: "border-hud-cyan/30", text: "text-hud-cyan", bg: "bg-hud-cyan/5" },
  warning: { border: "border-amber-500/30", text: "text-amber-400", bg: "bg-amber-500/5" },
}

function ToastItem({ toast }: { toast: Toast }) {
  const removeToast = useToastStore((s) => s.removeToast)
  const colors = COLORS[toast.type]
  return (
    <motion.div
      layout
      initial={{ x: 80, opacity: 0 }}
      animate={{ x: 0, opacity: 1 }}
      exit={{ x: 80, opacity: 0 }}
      transition={{ type: "spring", stiffness: 400, damping: 30 }}
      className={`flex items-center gap-2.5 px-3.5 py-2.5 rounded-xl border backdrop-blur-xl ${colors.border} ${colors.bg} shadow-lg max-w-xs`}
    >
      <span className={`shrink-0 ${colors.text}`}>{ICONS[toast.type]}</span>
      <span className="text-[11px] text-white/80 flex-1">{toast.message}</span>
      <button
        onClick={() => removeToast(toast.id)}
        className="shrink-0 text-white/20 hover:text-white/50 transition-colors"
      >
        <X className="h-3 w-3" />
      </button>
    </motion.div>
  )
}

export function ToastContainer() {
  const toasts = useToastStore((s) => s.toasts)
  return (
    <div className="fixed bottom-4 right-4 z-[9999] flex flex-col gap-2 items-end">
      <AnimatePresence mode="popLayout">
        {toasts.map((toast) => (
          <ToastItem key={toast.id} toast={toast} />
        ))}
      </AnimatePresence>
    </div>
  )
}
```

import { describe, it, expect, vi, beforeEach } from 'vitest'
import { useToastStore } from './toast'

describe('useToastStore', () => {
  beforeEach(() => {
    useToastStore.setState({ toasts: [] })
  })

  it('adds a toast', () => {
    useToastStore.getState().addToast('test message', 'info', 0)
    const toasts = useToastStore.getState().toasts
    expect(toasts).toHaveLength(1)
    expect(toasts[0].message).toBe('test message')
    expect(toasts[0].type).toBe('info')
  })

  it('removes a toast by id', () => {
    useToastStore.getState().addToast('to remove', 'info', 0)
    const id = useToastStore.getState().toasts[0].id
    useToastStore.getState().removeToast(id)
    expect(useToastStore.getState().toasts).toHaveLength(0)
  })

  it('deduplicates same message within 2s', () => {
    useToastStore.getState().addToast('dup', 'info', 0)
    useToastStore.getState().addToast('dup', 'info', 0)
    expect(useToastStore.getState().toasts).toHaveLength(1)
  })

  it('allows different messages', () => {
    useToastStore.getState().addToast('first', 'info', 0)
    useToastStore.getState().addToast('second', 'info', 0)
    expect(useToastStore.getState().toasts).toHaveLength(2)
  })

  it('auto-removes after duration', async () => {
    vi.useFakeTimers()
    useToastStore.getState().addToast('timed', 'info', 1000)
    expect(useToastStore.getState().toasts).toHaveLength(1)
    vi.advanceTimersByTime(1001)
    expect(useToastStore.getState().toasts).toHaveLength(0)
    vi.useRealTimers()
  })
})
```

"use client"
import { motion } from "framer-motion"
import { MapPin, Satellite, BarChart3, Search } from "lucide-react"

interface SuggestedPromptsProps {
  onSend: (message: string) => void
}

const SUGGESTIONS = [
  { icon: <MapPin className="h-4 w-4" />, text: "分析北京市学校分布" },
  { icon: <Satellite className="h-4 w-4" />, text: "计算NDVI植被指数" },
  { icon: <BarChart3 className="h-4 w-4" />, text: "生成人口密度热力图" },
  { icon: <Search className="h-4 w-4" />, text: "搜索成都市天府广场" },
]

export function SuggestedPrompts({ onSend }: SuggestedPromptsProps) {
  return (
    <div className="px-4 py-3 flex gap-2 overflow-x-auto">
      {SUGGESTIONS.map((s, i) => (
        <motion.button
          key={s.text}
          initial={{ opacity: 0, y: 10 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ delay: i * 0.05, duration: 0.2 }}
          onClick={() => onSend(s.text)}
          className="flex items-center gap-2 px-3 py-2 rounded-xl border border-white/[0.06] bg-white/[0.02] text-white/50 text-[11px] whitespace-nowrap hover:bg-hud-cyan/[0.06] hover:border-hud-cyan/20 hover:text-hud-cyan transition-all shrink-0"
        >
          {s.icon}
          {s.text}
        </motion.button>
      ))}
    </div>
  )
}
```
